type repository_error =
  1. | Not_found of {
    1. entity : string;
    2. id : Id.t;
    }
  2. | Already_exists of {
    1. entity : string;
    2. field : string;
    3. value : string;
    }
  3. | Constraint_violation of {
    1. entity : string;
    2. constraint_ : string;
    }
  4. | Database_error of string
  5. | Connection_lost
type validation_error =
  1. | Required_field of string
  2. | Invalid_format of {
    1. field : string;
    2. expected : string;
    3. got : string;
    }
  3. | Out_of_range of {
    1. field : string;
    2. min : int64;
    3. max : int64;
    4. value : int64;
    }
  4. | Duplicate of {
    1. entity : string;
    2. field : string;
    3. value : string;
    }
  5. | Invalid_enum of {
    1. field : string;
    2. valid_values : string list;
    3. got : string;
    }
  6. | Empty_string of string
  7. | Too_long of {
    1. field : string;
    2. max_length : int;
    3. length : int;
    }
type domain_error =
  1. | Item_not_available of {
    1. entity : string;
    2. name : string;
    3. reason : string;
    }
  2. | Item_already_checked_out of {
    1. entity : string;
    2. name : string;
    }
  3. | Insufficient_stock of {
    1. consumable : string;
    2. available : int;
    3. requested : int;
    }
  4. | Maintenance_required of {
    1. entity : string;
    2. name : string;
    3. reasons : string list;
    }
  5. | Invalid_reference of {
    1. entity : string;
    2. field : string;
    3. referenced_entity : string;
    4. ref_id : Id.t;
    }
  6. | Borrower_not_found of Id.t
  7. | Borrower_has_active_checkouts of Id.t
  8. | Circular_reference of {
    1. entity : string;
    2. field : string;
    }
type import_error =

  2. | Invalid_section of string
  3. | Missing_section of string

    2. row : int;
